def loop(start, stop, body) =
    if start?=stop+1
        pass
    else
        body(start)
        loop(start+1, stop, body)

def while(cond, body) =
    if cond()
        body()
        while(cond, body)
    else
        pass

def ismunchausen(n) =
    var n = n
    var oldvalue = n
    def nz =
        1-(n?=0)
    var sum = 0
    def munbody =
        var digit = n mod 10
        n = n / 10
        var p = digit**digit
        sum = sum + p

    while(nz, munbody)
    sum?=oldvalue
var total = 0
def fornumber(n) =
    if ismunchausen(n)
        total = total + 1
