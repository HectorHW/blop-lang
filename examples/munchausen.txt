def loop(start, stop, body) =
    if start?=stop+1
        0
    else
        body(start)
        loop(start+1, stop, body)

def while(cond, body) =
    if cond()
        body()
        while(cond, body)
    else
        0

def pow(a, b) =
    def pow(a, b, acc) =
        if b?=0
            acc
        else
            pow(a, b-1, acc*a)
    pow(a,b,1)

def ismunchausen(n) =
    var oldvalue = n
    def nz =
        1-(n?=0)
    var sum = 0
    def munbody =
        var digit = n mod 10
        n = n / 10
        var p = pow(digit, digit)
        sum = sum + p

    while(nz, munbody)
    sum?=oldvalue
var total = 0
def fornumber(n) =
    if ismunchausen(n)
        total = total + 1

loop(1, 5000, fornumber)
