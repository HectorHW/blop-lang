#you can add type annotations to variables
var a: Int = 2

print(a)

assert a==2

var some_global = 0

#and also function arguments
def F(a: Int, b: Int): Int =
    some_global = 1
    a+b

assert F(1,2)==3

#and also struct/enum variant fields
struct TextIndex:
    index: Int
    text: String

var instance = TextIndex(1, "abcd")
assert instance.index == 1
