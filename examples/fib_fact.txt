def fib(n) =
    if n<=1
        1
    else
        fib(n-1)+fib(n-2)

var fibvalue = fib(10)
assert fibvalue==89
print fibvalue

def fact(n) =
    def fact(n, a) =
        if n<=1
            a
        else
            fact(n-1, a*n)

    fact(n, 1)

var factvalue = fact(5)
assert factvalue==120
print factvalue
