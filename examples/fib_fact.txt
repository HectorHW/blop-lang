def fib(n) =
    if n?=0
        1
    else
        if n?=1
            1
        else
            fib(n-1)+fib(n-2)

var fibvalue = fib(10)
assert fibvalue?=89
print fib(10)

def fact(n) =
    if n?=1
        1
    else
        n*fact(n-1)
var factvalue = fact(5)
assert factvalue?=120
print factvalue
