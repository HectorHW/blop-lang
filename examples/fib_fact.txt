def fib(n) =
    if n==0
        1
    elif n==1
        1
    else
        fib(n-1)+fib(n-2)

var fibvalue = fib(10)
assert fibvalue==89
print fibvalue

def fact(n) =
    def fact1(n, a) =
        if n==1
            a
        else
            fact1(n-1, a*n)

    fact1(n, 1)

var factvalue = fact(5)
assert factvalue==120
print factvalue
